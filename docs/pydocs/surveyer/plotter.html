140    def make_table(self, questionids, alttitles = []):
141        """Produces table from text answers, deleting rows without answers."""
142
143        acceptedtypes = ["text", "enumerate", "select"]

156    def make_count_chart(self, questionid, charttype = "bar", title = "", altanswers = []):
157        """Produces a graphic to show basic statistic of selection or enumeration questions.
158        Chart type can be 'bar', 'pie' or 'line', and you can provide alternative title and answer lables."""
159
160        acceptedtypes = ["select", "enumerate"]
161
162        if not title:
163            title = "Count of responses"
164        
165        df_extract = self.dataset.extract_subset([questionid], acceptedtypes)
166
167        if self.dataset.metadata["A2"]["entrytype"] == "enumerate":
168            df_nona = df_extract.dropna()
169            df_enlisted = [entry.split("; ") for entry in df_nona[df_extract.columns[0]]]
170            df_extract = pd.DataFrame({df_extract.columns[0]: [item for sublist in df_enlisted for item in sublist]})
171        
172        answer_counts = df_extract.value_counts()
173    
174        #create the bar chart
175        plt.figure(figsize=(6, 4))
176        if charttype != "pie":
177            answer_counts.plot(kind=charttype)
178            plt.xlabel('Response')
179            plt.ylabel('Frequency')
180            plt.xticks(rotation=40, ha='right')
181        else:
182            plt.xlabel('Number of entries: '+str(answer_counts.sum()))
183    
184        plt.title(title)
185
186        if altanswers:
187            if len(answer_counts) == len(altanswers):
188                if charttype != "pie":
189                    plt.xticks(ticks=range(len(altanswers)), labels=altanswers, rotation=40, ha='right')
190                else:
191                    plt.pie(answer_counts, labels=altanswers, autopct='%1.1f%%', startangle=90)
192            else:
193                print ("Alternative options length", len(altanswers), "does not match required length", len(answer_counts))
194    
195        filename = "plot_"+charttype + "_" + str(questionid).replace("[", "").replace("]", "").replace(",", "_").replace("'", "")  + ".png"
196
197        plt.tight_layout()
198        # Save the chart as an image
199        plt.savefig(filename)
200
201        return filename

203    def make_compare_likert(self, questionid, subquestion, title = "", altanswers = []):
204        """Displays several Likert-scale type answers for comparison. 
205           Needs subquestion to select all subquestions which contain the provided string.
206           Can be passed title for the plot and answer options to replace the original ones.
207           Scales can be provided under the 'scales' entry in the configuration file.
208        """
209        acceptedtypes = ["select"]
210    
211        if not title:
212            title = "Estimate"
213        
214        #df_extract = self.dataset.extract_subset(questionids, acceptedtypes) # also extract subquestion
215
216        columns = []
217        altnames = []
218        for i in range(len(self.dataset.metadata[questionid]["subquestions"])):
219            otherpart = []
220            foundpart = ""
221            for entry in self.dataset.metadata[questionid]["subquestions"][i]:    
222                if entry.find(subquestion)>-1:
223                    foundpart = self.dataset.metadata[questionid]["colnames"][i]
224                else:
225                    otherpart.append(entry)
226            if foundpart:
227                columns.append(foundpart)
228                altnames.append(otherpart)   
229
230        if altanswers:
231            altnames = altanswers
232        else:
233            altnames = [str(ent).replace("['","").replace("']","").replace("','"," : ") for ent in altnames]
234        
235        plottype = ""
236        options = ""
237    
238        for col in columns:
239            if col in self.dataset.metadata[questionid]["params"]["options"]: 
240                if not options:
241                    options = self.dataset.metadata[questionid]["params"]["options"][col]
242                else:
243                    newopts = self.dataset.metadata[questionid]["params"]["options"][col]
244                    for opt in newopts:
245                        if not opt in options:
246                            options.append(opt)                    
247                        
248            if col in self.dataset.metadata[questionid]["params"]["subtypes"]:
249                if not plottype:
250                    plottype = self.dataset.metadata[questionid]["params"]["subtypes"][col]
251                else:
252                    if self.dataset.metadata[questionid]["params"]["subtypes"][col] != plottype:
253                        print ("Question types don't match!",self.dataset.metadata[questionid]["params"]["subtypes"][col], plottype)
254                        plottype = "mismatch"
255    
256        scale = []
257
258        if "scales" in self.conf:
259            for skey in self.conf["scales"]:
260                if sorted(self.conf["scales"][skey]["options"]) == sorted(options):
261                    scale = self.conf["scales"][skey]["options"]
262                
263        if not scale:
264            scale = options
265    
266        df = self.dataset.data[columns]
267    
268        df_melted = df.melt(var_name=subquestion, value_name='Response')
269    
270        # Plot the grouped bar chart
271        plt.figure(figsize=(12, 6))
272        sns.countplot(data=df_melted, x=subquestion, hue='Response', order=df.columns, hue_order=scale, palette='coolwarm')
273        
274        # Add titles and labels
275        plt.title(title)
276        plt.xlabel('Question')
277        plt.ylabel('Count')
278        plt.legend(title=subquestion, bbox_to_anchor=(1.05, 1), loc='upper left')  # Move the legend outside
279        plt.xticks(ticks=range(len(altnames)), labels=altnames)  # Rotate x-axis labels if neede
280    
281        plt.tight_layout()  # Adjust layout for readability
282    
283        filename = "plot_"+subquestion.replace(" ", "_") + "_" + str(questionid) + ".png"
284    
285        plt.savefig(filename)
286
287        return filename

289    def make_rating(self, questionid, title = ""):
290        """Displays distribution of a rating question."""
291    
292        acceptedtypes = ["rating", "select"]
293    
294        df_extract_all = self.dataset.extract_subset([questionid])
295        df_extract = df_extract_all[df_extract_all.columns[0]]
296       
297        all_ratings = list(range(1, self.dataset.metadata[questionid]["params"]["options"]["factor"]+1))
298    
299        df_extract = pd.DataFrame({df_extract_all.columns[0]: [int(entry.split("/")[0]) for entry in df_extract]}).dropna()
300        df_extract = pd.to_numeric(df_extract[df_extract_all.columns[0]], errors='coerce')
301        rating_counts = df_extract.value_counts().reindex(all_ratings, fill_value=0)
302        
303        # Calculate Mean and RMS
304        mean_rating = df_extract.mean()
305        
306        # Plot the distribution of ratings
307        plt.figure(figsize=(8, 6))
308        ax = sns.barplot(x=rating_counts.index, y=rating_counts.values, palette="viridis")
309        
310        # Add Mean and RMS to the legend
311        mean_label = f'Mean: {mean_rating:.2f}'
312        handles, labels = ax.get_legend_handles_labels()
313        handles.extend([plt.Line2D([0], [0], color='none', label=mean_label)])
314        ax.legend(handles=handles, loc='upper right')
315    
316        if not title:
317            title = df_extract_all.columns[0]
318        
319        # Adding titles and labels
320        plt.title(title)
321        plt.xlabel("Rating")
322        plt.ylabel("Count")
323
324        filename = "plot_rating_"+ str(questionid) + ".png"
325    
326        plt.savefig(filename)
327
328        return filename

330    def make_multirating(self, questionid, title = "", excludeterms = []):
331
332        columnsall = self.dataset.metadata[questionid]["colnames"]
333
334        columns = []
335        for col in columnsall:
336            dontuse = False
337            for term in excludeterms:
338                if col.find(term) > -1:
339                    dontuse = True
340            if not excludeterms or not dontuse:
341                columns.append(col)
342
343        df = self.dataset.data[columns].dropna()
344        df[columns] = df[columns].applymap(lambda x: int(x.split('/')[0]))
345    
346        altnames = [str(entry[0]) for entry in self.dataset.metadata[questionid]["subquestions"]]
347    
348        renamemap = {}
349        for i in range(len(columnsall)):
350            dontuse = False
351            for term in excludeterms:
352                if columnsall[i].find(term) > -1:
353                    dontuse = True
354            if not excludeterms or not dontuse:
355                renamemap.setdefault(columnsall[i], altnames[i])
356        df = df.rename(columns=renamemap)
357        
358        # Calculate mean and standard deviation for each question
359        means = df.mean()
360        stds = df.std()
361    
362        # Plot each question as a horizontal bar with error bars for standard deviation
363        plt.figure(figsize=(8, 6))
364        plt.barh(y=means.index, width=means, xerr=stds, color='skyblue', edgecolor='gray')
365        
366        # Add mean values next to bars for clarity
367        for index, value in enumerate(means):
368            plt.text(value + 0.1, index, f'{value:.2f}', va='center', color='black')
369    
370        if not title:
371            title = "Mean and Spread of Ratings by Question"
372        # Add labels and title
373        plt.xlabel("Rating out of "+str(self.dataset.metadata[questionid]["params"]["options"]["factor"]))
374        plt.title(title)
375        plt.grid(axis='x', linestyle='--', alpha=0.7)
376
377        plt.tight_layout()
378
379        filename = "plot_multirating_"+ str(questionid) + ".png"
380    
381        plt.savefig(filename)
382
383        return filename
